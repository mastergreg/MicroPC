PRINT_NUM MACRO CHAR  
    PUSH DX
    PUSH AX
    MOV DX,CHAR      
    ADD DX,30H
    MOV AH,2
    INT 21H
    POP AX
    POP DX
    
ENDM

PRINT MACRO CHAR
    PUSH AX
    PUSH DX
    MOV DL,CHAR
    MOV AH,02H
    INT 21H
    POP DX
    POP AX
ENDM

PRINT_STRING MACRO STRING
    PUSH AX
    PUSH DX
    MOV DX,OFFSET STRING  ;ASSUME THAT STRING IS A VARIABLE OR CONSTANT, not AN ADDRESS
    MOV AH,09H
    INT 21H
    POP DX
    POP AX
Endm
