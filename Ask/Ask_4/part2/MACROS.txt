;This macro change registers AH,AL
READ MACRO 
    MOV AH,1
    INT 21H
ENDM  

READNB MACRO 
    MOV AH,06H
    MOV DL,0FFH
    INT 21H
ENDM  

;This macro uses registers AH,DL
PRINT MACRO CHAR
    PUSH AX
    PUSH DX
    MOV DL,CHAR
    MOV AH,02H
    INT 21H
    POP DX
    POP AX
ENDM


; This will use INT 10/02H
; to move the cursor 
; to the appropriate position
PRINT_ECHO MACRO CHAR
    PUSH AX
    PUSH DX
    MOV AH,02H
    MOV BP,OFFSET ECHO_LINE
    MOV DH,DS:[BP]
    MOV BP,OFFSET ECHO_COL
    MOV DL,DS:[BP]
    MOV BH,0H
    INT 10H
    MOV AL,CHAR
    MOV AH,0EH
    INT 10H
    POP DX
    POP AX
ENDM

;This macro uses registers AH,DX
PRINT_STRING MACRO STRING
    PUSH AX 
    PUSH DX
    MOV DX,OFFSET STRING ;Assume that string is a variable or constant, NOT an address
    MOV AH,09H
    INT 21H 
    POP DX
    POP AX
ENDM

EXIT MACRO
    MOV AH,4CH
    INT 21H
ENDM
