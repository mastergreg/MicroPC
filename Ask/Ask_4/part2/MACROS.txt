;This macro change registers AH,AL
READ MACRO 
    MOV AH,1
    INT 21H
ENDM  

READNB MACRO 
    MOV AH,06H
    MOV DL,0FFH
    INT 21H
ENDM  

;This macro uses registers AH,DL
PRINT MACRO CHAR
    PUSH AX
    PUSH DX
    MOV DL,CHAR
    MOV AH,02H
    INT 21H
    POP DX
    POP AX
ENDM

;This macro uses registers AH,DX
PRINT_STRING MACRO STRING
    PUSH AX 
    PUSH DX
    MOV DX,OFFSET STRING ;Assume that string is a variable or constant, NOT an address
    MOV AH,09H
    INT 21H 
    POP DX
    POP AX
ENDM

EXIT MACRO
    MOV AH,4CH
    INT 21H
ENDM
